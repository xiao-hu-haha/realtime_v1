###   分析倾斜问题
```java
with t1 as (
    select userid,
           order_no, 
           concat(region,'-',rand()) as region,
           product_no,
           color_no,
           sale_amount,
           ts  
    from date_east
//主要把数据打散        
),
    t2 as (
        select region,
               count(1) as cnt
        from t1
        group by region
 //  聚合       
    ),
    t3 as (
        select region,
               substr(region,1,2) as re,
               cnt
        from t2
 //  截取       
    ),
    t4 as (
        select re,
               count(1) as cnt
        from t3
        group by re
//    分组聚合      
    )
select /*+ mapjoin(dim_region_east) */
       a.*,
       b.region_id
from t4 as a
join dim_region_east as b
on a.re = b.region_name;
 //  mapjoin一下，把小表放在前面，大表在后面，解决倾斜问题      
```
